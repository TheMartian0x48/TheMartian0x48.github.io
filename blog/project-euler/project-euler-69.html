Project Euler : Problem 69

Link

https://projecteuler.net/problem=69

Hints

Euler's totient function, phi of range[1, n] in O(n * loglog(n))

Source Code

[ C++ ]

#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 1;
vector<int> phi(N);

void GeneratePhi() {
  iota(phi.begin(), phi.end(), 0);
  for (int i = 2; i < N; i += 2) {
    phi[i] -= phi[i] >> 1;
  }
  for (int i = 3; i < N; i += 2) {
    if (phi[i] == i) {
      for (int j = i; j < N; j += i) {
        phi[j] -= phi[j] / i;
      }
    }
  }
}


int main() {

  GeneratePhi();
  long long num = 0, deno = 1;
  int result = 0;

  for (int i = 2; i < N; i++) {
    long long num1 = num * phi[i];
    long long num2 = deno * i;
    if (num2 > num1) {
      num = i, deno = phi[i];
      result = i;
    }
  }
  cout << result;
  return 0;
}
