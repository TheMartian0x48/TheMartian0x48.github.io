Hints

Source Code

[ C++ ]

#include <bits/stdc++.h>
using namespace std;

int main() {
  vector<int> fac(10, 1);
  for (int i = 1; i < 10; i++) {
    fac[i] = fac[i - 1] * i;
  }
  int res = 0;
  for (int i = 3; i <= 5000000; i++) {
    int sum = 0, n = i;
    while(n) {
      sum += fac[n % 10];
      n /= 10;
    }
    if (i == sum) res += i;
  }
  cout << res;
  return 0;
}
