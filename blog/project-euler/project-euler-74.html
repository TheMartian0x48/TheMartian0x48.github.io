Hints

Source Code

[ C++ ]

#include <bits/stdc++.h>
using namespace std;

int main() {
  vector<int> fac(10, 1);
  for (int i = 1; i < 10; i++) {
    fac[i] = fac[i - 1] * i;
  }

  vector<int> chains(3000000, 0);
  unordered_set<int> seen;
  int res = 0;
  for (int i = 1; i <= 1000000; i++) {
    seen.clear();
    int n = i;
    bool leave = false;
    while (seen.count(n) == 0) {
      if (chains[n] != 0) {
        chains[i] = seen.size() + chains[n];
        leave = true;
        break;
      }
      seen.insert(n);
      int t = n;
      n = 0;
      while (t) {
        n += fac[t % 10];
        t /= 10;
      }
    }
    if (!leave) {
      chains[i] = seen.size();
    }
    res += chains[i] == 60;
  }
  cout << res;
  return 0;
}
