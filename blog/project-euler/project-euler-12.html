#include <bits/stdc++.h>
using namespace std;

const int NMAX = 1e6;
vector<int> spf(NMAX);
vector<int> primes;

// Calculate smallest primes factor for each x in (1, n)
void CalculateSPF() {
  iota(spf.begin(), spf.end(), 0);
  for (int i = 4; i < NMAX; i += 2)
    spf[i] = 2;
  primes.push_back(2);
  for (long long i = 3; i < NMAX; i += 2) {
    if (spf[i] == (int)i) {
      for (long long j = i * i; j < (long long)NMAX; j += i) {
        if (spf[j] == j) {
          spf[j] = i;
        }
      }
      primes.push_back(i);
    }
  }
}

// Calculate number of factor
int NumberOfFactor(int n) {
  if (n < 2) return 1;
  int res = 1;
  // handle the case when n >= NMAX
  for (int i = 0; i < (int)primes.size() && n >= NMAX; i++) {
    int p = primes[i];
    int c = 0;
    while (n % p == 0) {
      c++;
      n /= p;
    }
    res *= (c + 1);
  }
  // if n is prime number >= NMAX
  if (n >= NMAX) {
    return 2; 
  }
  while (n != 1) {
    int p = spf[n];
    int c = 0;
    while (n != 1 && spf[n] == p) {
      c++;
      n /= p;
    }
    res *= (c + 1);
  }
  return res;
}

int main() {
  CalculateSPF();
  int s = 1, i = 2;
  while (NumberOfFactor(s) < 500 ) {
    s += i++;
  }
  cout << s << endl;
}
