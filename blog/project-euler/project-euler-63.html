Source Code

[ C++ ]

#include <bits/stdc++.h>
using namespace std;

int main() {
  vector<long double> log10_val(9);
  for (int i = 1; i < 10; i++) {
    log10_val[i - 1] = log10(i);
  }
  int res = 0;
  for (int i = 1;; i++) {
    long double lb = (long double)(i - 1) / i;
    int c = log10_val.end() - lower_bound(log10_val.begin(), log10_val.end(), lb);
    if (c == 0) break;
    else res += c;
  }
  cout << res << endl;
}
