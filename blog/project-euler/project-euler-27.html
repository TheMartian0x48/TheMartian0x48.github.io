#include <bits/stdc++.h>
using namespace std;

// return all primes in range (1, n)
vector<long long> GetPrimes(long long n) {
  vector<bool> is_prime(n, true);
  vector<long long> res;
  res.push_back(2);
  // discard all even number > 2
  for (long long i = 4; i < n; i += 2)
    is_prime[i] = false;
  // only check for odd number
  for (long long i = 3; i < n; i += 2) {
    if (is_prime[i]) {
      for ( long long j = i * i; j < n; j += i) {
        is_prime[j] = false;
      }
      res.push_back(i);
    }
  }
  return res;
}

int main() {
  auto primes = GetPrimes(100000);
  unordered_set<int> prime;
  prime.reserve(100000);
  for (auto a : primes) {
    prime.insert(a);
  }
  long long res = 0;
  int mx = 0;
  for (long long a = -999; a < 1000; a++) {
    for (long long b = -999; b < 1000; b++) {
      int n = 0;
      while (true) {
        long long tmp = n * n + a * n + b;
        if (prime.count(tmp) == 0) {
          n--;
          break;
        }
        n++;
      }
      if (n > mx) {
        mx = n;
        res = a * b;
      }
    }
  }
  cout << res;
  return 0;
}
