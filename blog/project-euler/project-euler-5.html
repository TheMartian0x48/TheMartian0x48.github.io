#include <bits/stdc++.h>
using namespace std;

int main() {
  long long res = 1;
  for (long long i = 1; i <= 20; i++) {
    long long g = __gcd(res, i);
    res = res / g * i;
  }
  cout << res << endl;
  return 0;
}
