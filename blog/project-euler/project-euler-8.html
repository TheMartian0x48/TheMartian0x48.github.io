#include <bits/stdc++.h>
using namespace std;

int main() {
  string n;
  for (int i = 0; i < 20; i++) {
    string tmp; cin >> tmp;
    n += tmp;
  }
  int len = n.size();
  long long product = 1, result = 0;
  for (int i = 0, j = 0; i < len; i++) {
    product *= (n[i] - '0');
    // case when product = 0
    if (product == 0) {
      j = i + 1;
      product = 1;
      continue;
    }
    // decrease sliding window when sliding window greater than 13
    if (i - j + 1 > 13) {
      product /= (n[j++] - '0');
    }
    if (i - j + 1 == 13) {
      result = max(result, product);
    }
  }
  cout << result << endl;
  return 0;
}
