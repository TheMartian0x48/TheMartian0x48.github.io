#include <bits/stdc++.h>
using namespace std;

// return all primes in range (1, n)
vector<long long> GetPrimes(long long n) {
  vector<bool> is_prime(n, true);
  vector<long long> res;
  res.push_back(2);
  // discard all even number > 2
  for (long long i = 4; i < n; i += 2)
    is_prime[i] = false;
  // only check for odd number
  for (long long i = 3; i < n; i += 2) {
    if (is_prime[i]) {
      for ( long long j = i * i; j < n; j += i) {
        is_prime[j] = false;
      }
      res.push_back(i);
    }
  }
  return res;
}

int main() {
  long long n = 600851475143;
  long long ub = sqrt(n) + 1, res = n;
  vector<long long> primes = GetPrimes(ub);
  for (int i = primes.size() - 1; i >= 0; i--) {
    long long p = primes[i];
    if (n % p == 0) {
      res = p;
      break; // this is greatest 
    }
  }
  cout << res << endl;
}
