Project Euler : Problem 73

Link

https://projecteuler.net/problem=73

Hints

Co-prime, GCD, Fraction

Source Code

[ C++ ]

#include <bits/stdc++.h>
using namespace std;

bool cmp(const pair<int, int>&p1,const pair<int, int>&p2) {
  long long n1 = 1LL * p1.first * p2.second;
  long long n2 = 1LL * p2.first * p1.second;
  return n1 < n2;
}

int main() {
  const int n = 12000;
  set<pair<int, int>> fractions;
  pair<int, int> lb = {1, 3}, ub = {1, 2};

  for (int i = 1; i <= n; i++) {
    for (int j = 1; j < i; j++) {
      int g = __gcd(i, j);
      pair<int, int> fraction {j / g, i / g};
      if (cmp(fraction, lb)) continue;
      if (cmp(lb, fraction)  == false) continue;
      if (cmp(fraction, ub)) {
        fractions.insert(fraction);
      }
    }
  }
  cout << fractions.size() << endl;
  return 0;
}
