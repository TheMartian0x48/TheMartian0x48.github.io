Project Euler : Problem 16

Link

https://projecteuler.net/problem=16

Hints

Big integer, $2^{n + 1} = 2^{n} + 2^{n}$

Source Code

[ C++ ]

#include <bits/stdc++.h>
using namespace std;

string Add(const string &a, const string &b) {
  string result{};
  int i = a.size() - 1, j = b.size() - 1, carry = 0;
  while (i >= 0 || j >= 0) {
    if (i >= 0) carry += a[i--] - '0';
    if (j >= 0) carry += b[j--] - '0';

    result += '0' + carry % 10;
    carry /= 10;
  }
  if (carry)
    result += '0' + carry;
  reverse(result.begin(), result.end());
  return result;
}


int main() {
  string n = "1";
  for (int i = 1; i <= 1000; i++) {
    n = Add(n, n);
  }  
  int sum = 0;
  for (auto d : n) {
    sum += d - '0';
  }
  cout << sum;
  return 0;
}
